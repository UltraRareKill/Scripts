-- ── RESIZE ───────────────────────────────────────────────────────
local resizing, resizeStart, sizeStart = false, nil, nil

grip.InputBegan:Connect(function(i)
    if i.UserInputType == Enum.UserInputType.MouseButton1 then
        resizing    = true
        resizeStart = i.Position
        sizeStart   = Vector2.new(main.Size.X.Offset, main.Size.Y.Offset)
    end
end)

grip.InputEnded:Connect(function(i)
    if i.UserInputType == Enum.UserInputType.MouseButton1 then
        resizing = false
        if not isMinimized then
            savedSize = main.Size
        end
    end
end)

UserInputService.InputChanged:Connect(function(i)
    if i.UserInputType ~= Enum.UserInputType.MouseMovement then return end

    if dragging then
        local d = i.Position - dragStart
        main.Position = UDim2.new(
            winStart.X.Scale, winStart.X.Offset + d.X,
            winStart.Y.Scale, winStart.Y.Offset + d.Y
        )
    end

    if resizing then
        local d    = i.Position - resizeStart
        local newW = math.clamp(sizeStart.X + d.X, MIN_W, MAX_W)
        local newH = math.clamp(sizeStart.Y + d.Y, MIN_H, MAX_H)
        main.Size  = UDim2.new(0, newW, 0, newH)
    end
end)

-- ── SEND MESSAGE ─────────────────────────────────────────────────
local function sendMessage()
    if isBusy then return end
    local text = inputBox.Text:match("^%s*(.-)%s*$")
    if text == "" then return end

    if API_KEY == "" then
        addBubble("⚠  Please set your Groq API key first — click ⚙ Settings.", false)
        return
    end

    isBusy = true
    inputBox.Text            = ""
    sendBtn.BackgroundColor3 = C.winbtn
    sendBtn.Text             = "…"

    addBubble(text, true)
    table.insert(messages, {role = "user", content = text})

    local typing = addTyping()

    task.spawn(function()
        local reply
        for attempt = 1, 3 do
            local ok, raw = pcall(function()
                return HttpService:PostAsync(
                    URL,
                    HttpService:JSONEncode({messages = messages, apiKey = API_KEY}),
                    Enum.HttpContentType.ApplicationJson
                )
            end)

            if ok then
                local data = HttpService:JSONDecode(raw)
                if data.choices then
                    reply = data.choices[1].message.content
                    break
                elseif data.error then
                    reply = "⚠  " .. (data.error.message or "Unknown API error.")
                    break
                end
            else
                local err = tostring(raw):lower()
                if err:find("http requests are not enabled") or err:find("httpenabled") then
                    reply = "❌  HTTP is disabled in this game!\n\nGo to: Game Settings → Security → Enable 'Allow HTTP Requests', then rejoin."
                    break
                elseif err:find("connection") or err:find("timeout") or err:find("ssl") then
                    reply = "⚠  Cannot connect to server. Make sure the Replit server is running."
                    break
                end
                -- Otherwise retry
            end
            task.wait(2)
        end

        if typing.Parent then typing:Destroy() end
        reply = reply or "⚠  No response after 3 attempts. Check your connection."
        table.insert(messages, {role = "assistant", content = reply})
        addBubble(reply, false)

        isBusy                   = false
        sendBtn.BackgroundColor3 = C.accent
        sendBtn.Text             = "↑"
    end)
end

sendBtn.MouseButton1Click:Connect(sendMessage)

inputBox.FocusLost:Connect(function(entered)
    if entered then sendMessage() end
end)
